CREATE TABLE Cliente (
    ID_CLIENTE NUMBER NOT NULL,
    Nombre VARCHAR2(12) NOT NULL,
    Apellidos VARCHAR2(50) NOT NULL,
    Direccion VARCHAR2(50) NOT NULL,
    Telefono VARCHAR2(15) NOT NULL,
    Metodo_pago VARCHAR2(25) NOT NULL,
    CONSTRAINT PK_ID_CLIENTE PRIMARY KEY(ID_CLIENTE));

CREATE TABLE Factura (
    ID_PEDIDO NUMBER NOT NULL,
    ID_CLIENTE NUMBER NOT NULL,
    FECHA_FACTURA DATE,
    CONSTRAINT PK_ID_PEDIDO PRIMARY KEY (ID_PEDIDO),
    CONSTRAINT FK_ID_CLIENTE FOREIGN KEY (ID_CLIENTE) REFERENCES Cliente (ID_CLIENTE)
);

CREATE TABLE Productos (
    ID_PRODUCTO NUMBER NOT NULL,
    NOMBRE_PRODUCTO VARCHAR2(50) NOT NULL,
    DESCRIPCION_PRODUCTO VARCHAR2(100) NOT NULL,
    CATEGORIA_PRODUCTO VARCHAR2(50) NOT NULL,
    PRECIO_PRODUCTO NUMBER (5,2) NOT NULL,
    CONSTRAINT PK_ID_PRODUCTO PRIMARY KEY (ID_PRODUCTO)
);

CREATE TABLE Reservas (
    ID_RESERVA NUMBER NOT NULL,
    ID_CLIENTE NUMBER NOT NULL,
    FECHA_HORA_RESERVA DATE NOT NULL,
    NUMERO_PERSONAS NUMBER,
    ESTADO_RESERVA VARCHAR2(20) NOT NULL,
    CONSTRAINT PK_ID_RESERVA PRIMARY KEY (ID_RESERVA),
    CONSTRAINT FK_ID_CLIENTE_RESERVAS FOREIGN KEY (ID_CLIENTE) REFERENCES Cliente (ID_CLIENTE)
);

CREATE TABLE Estado_pedido
    PK ID ESTADO PEDIDO
CONSTRAINT FK_ID_CLIENTE_PEDIDO FOREIGN KEY (ID_CLIENTE) REFERENCES Cliente (ID_CLIENTE)
CONSTRAINT FK_ID_ESTADOPEDIDO_PEDIDO FOREIGN KEY (ID_CLIENTE) REFERENCES Cliente (ID_CLIENTE)
